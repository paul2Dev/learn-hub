Pasul 1 — Supabase: schema și RLS

-- În Supabase SQL Editor:
CREATE TABLE tasks (
  id          UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id     UUID    REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,
  title       TEXT    NOT NULL CHECK (char_length(title) > 0),
  description TEXT,
  done        BOOLEAN DEFAULT false NOT NULL,
  priority    TEXT    DEFAULT 'medium' CHECK (priority IN ('low', 'medium', 'high')),
  due_date    DATE,
  created_at  TIMESTAMPTZ DEFAULT now() NOT NULL,
  updated_at  TIMESTAMPTZ DEFAULT now() NOT NULL
);

-- RLS: activat + politici
ALTER TABLE tasks ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Utilizatorii văd numai task-urile lor" ON tasks
  FOR ALL USING (auth.uid() = user_id);

-- Setează user_id automat la INSERT
CREATE OR REPLACE FUNCTION set_user_id()
RETURNS TRIGGER AS $$
BEGIN
  NEW.user_id = auth.uid();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER tasks_set_user_id
  BEFORE INSERT ON tasks
  FOR EACH ROW EXECUTE FUNCTION set_user_id();

Pasul 4 — Server Routes API

// server/api/tasks.get.ts
import { serverSupabaseUser, serverSupabaseClient } from '#supabase/server'

export default defineEventHandler(async (event) => {
  const user = await serverSupabaseUser(event)
  if (!user) throw createError({ statusCode: 401, message: 'Neautorizat' })

  const client = await serverSupabaseClient(event)
  const { data, error } = await client
    .from('tasks')
    .select('*')
    .order('created_at', { ascending: false })

  if (error) throw createError({ statusCode: 500, message: error.message })
  return data
})
// server/api/tasks.post.ts
import { serverSupabaseUser, serverSupabaseClient } from '#supabase/server'

export default defineEventHandler(async (event) => {
  const user = await serverSupabaseUser(event)
  if (!user) throw createError({ statusCode: 401, message: 'Neautorizat' })

  const body = await readBody(event)
  if (!body?.title?.trim()) {
    throw createError({ statusCode: 400, message: 'Titlul este obligatoriu' })
  }

  const client = await serverSupabaseClient(event)
  const { data, error } = await client
    .from('tasks')
    .insert({ title: body.title.trim(), description: body.description, priority: body.priority ?? 'medium' })
    .select()
    .single()

  if (error) throw createError({ statusCode: 500, message: error.message })
  return data
})
// server/api/tasks/[id].patch.ts
import { serverSupabaseUser, serverSupabaseClient } from '#supabase/server'

export default defineEventHandler(async (event) => {
  const user = await serverSupabaseUser(event)
  if (!user) throw createError({ statusCode: 401, message: 'Neautorizat' })

  const id   = getRouterParam(event, 'id')
  const body = await readBody(event)

  const client = await serverSupabaseClient(event)
  const { data, error } = await client
    .from('tasks')
    .update({ ...body, updated_at: new Date().toISOString() })
    .eq('id', id)
    .select()
    .single()

  if (error) throw createError({ statusCode: 500, message: error.message })
  return data
})
// server/api/tasks/[id].delete.ts
import { serverSupabaseUser, serverSupabaseClient } from '#supabase/server'

export default defineEventHandler(async (event) => {
  const user = await serverSupabaseUser(event)
  if (!user) throw createError({ statusCode: 401, message: 'Neautorizat' })

  const id     = getRouterParam(event, 'id')
  const client = await serverSupabaseClient(event)
  const { error } = await client.from('tasks').delete().eq('id', id)

  if (error) throw createError({ statusCode: 500, message: error.message })
  return { success: true }
})

Pasul 5 — Pinia Store

// stores/tasks.ts
import { defineStore } from 'pinia'

interface Task {
  id: string
  title: string
  description: string | null
  done: boolean
  priority: 'low' | 'medium' | 'high'
  due_date: string | null
  created_at: string
}

export const useTasksStore = defineStore('tasks', () => {
  const tasks   = ref<Task[]>([])
  const loading = ref(false)

  const pending   = computed(() => tasks.value.filter(t => !t.done))
  const completed = computed(() => tasks.value.filter(t => t.done))

  async function fetchTasks() {
    loading.value = true
    tasks.value   = await $fetch('/api/tasks')
    loading.value = false
  }

  async function addTask(payload: Pick<Task, 'title'> & Partial<Pick<Task, 'description' | 'priority'>>) {
    const newTask = await $fetch('/api/tasks', { method: 'POST', body: payload })
    tasks.value.unshift(newTask)
  }

  async function toggleTask(id: string) {
    const task = tasks.value.find(t => t.id === id)
    if (!task) return
    task.done = !task.done  // optimistic update
    try {
      await $fetch(`/api/tasks/${id}`, { method: 'PATCH', body: { done: task.done } })
    } catch {
      task.done = !task.done  // rollback dacă eșuează
    }
  }

  async function deleteTask(id: string) {
    tasks.value = tasks.value.filter(t => t.id !== id)  // optimistic remove
    try {
      await $fetch(`/api/tasks/${id}`, { method: 'DELETE' })
    } catch {
      await fetchTasks()  // refetch dacă eșuează
    }
  }

  return { tasks, loading, pending, completed, fetchTasks, addTask, toggleTask, deleteTask }
})

Pasul 6 — Dashboard Page

<!-- pages/dashboard/index.vue -->
<script setup lang="ts">
definePageMeta({ middleware: 'auth' })  // sau prin supabase redirect automat

const store  = useTasksStore()
const user   = useSupabaseUser()

const newTitle = ref('')
const filter   = ref<'all' | 'pending' | 'done'>('all')

const filteredTasks = computed(() => {
  if (filter.value === 'pending')  return store.pending
  if (filter.value === 'done')     return store.completed
  return store.tasks
})

onMounted(() => store.fetchTasks())

async function handleSubmit() {
  if (!newTitle.value.trim()) return
  await store.addTask({ title: newTitle.value.trim() })
  newTitle.value = ''
}
</script>

<template>
  <div class="dashboard">
    <header>
      <h1>Task-urile mele</h1>
      <p>{{ store.completed.length }} / {{ store.tasks.length }} completate</p>
    </header>

    <!-- Formular adăugare -->
    <form @submit.prevent="handleSubmit">
      <input v-model="newTitle" placeholder="Task nou..." required />
      <button type="submit">Adaugă</button>
    </form>

    <!-- Filtre -->
    <div class="filters">
      <button :class="{ active: filter === 'all' }"     @click="filter = 'all'">Toate</button>
      <button :class="{ active: filter === 'pending' }" @click="filter = 'pending'">În așteptare</button>
      <button :class="{ active: filter === 'done' }"    @click="filter = 'done'">Completate</button>
    </div>

    <!-- Lista task-uri -->
    <div v-if="store.loading">Se încarcă...</div>
    <ul v-else>
      <li v-for="task in filteredTasks" :key="task.id">
        <input
          type="checkbox"
          :checked="task.done"
          @change="store.toggleTask(task.id)"
        />
        <span :class="{ done: task.done }">{{ task.title }}</span>
        <button @click="store.deleteTask(task.id)">✕</button>
      </li>
    </ul>
  </div>
</template>

Verifică-ți înțelegerea

Cinci întrebări finale. Dă clic pe un răspuns — feedback-ul este imediat.

Întrebarea 1 din 5 Ce este un optimistic update și de ce îl folosim în toggleTask?
Corect. Optimistic update înseamnă: modifici state-ul local imediat (UI reacționează instant), faci request-ul API în background, și dacă API eșuează, dai rollback. Rezultatul: aplicația pare mai rapidă (0ms lag vizual), chiar dacă internaly tot există latența de rețea.
Întrebarea 2 din 5 De ce validăm input-ul în server route, deși avem validare și în componenta Vue?
Corect. Validarea pe client (Vue) este pentru UX — feedback rapid, fără round-trip. Validarea pe server este pentru securitate — oricine poate trimite un request direct la /api/tasks cu Postman sau curl, ocolind complet UI-ul. Niciodată nu te baza exclusiv pe validarea clientului.
Întrebarea 3 din 5 În Pinia, care este diferența dintre ref() (state) și computed() (getter)?
Corect. ref(tasks) este state-ul sursă — îl modifici din actions. computed(() => tasks.filter(...)) este un getter derivat — se recalculează automat când tasks se schimbă, este cached, și este read-only. Nu ai nevoie să gestionezi sincronizarea manual.
Întrebarea 4 din 5 Dacă userul șterge un task și DELETE API returnează eroare, ce face store-ul nostru?
Corect. În loc să stocăm task-ul temporar pentru rollback, facem refetch complet din Supabase — mai simplu și mai robust. Dezavantajul: un request extra. Avantajul: state-ul este garantat sincronizat cu baza de date, indiferent ce s-a întâmplat în spate.
Întrebarea 5 din 5 RLS (Row Level Security) pe tabela tasks ce garantează?
Corect. RLS este securitate la nivel de bază de date — nu la nivel de API sau UI. Chiar dacă un utilizator obține cumva URL-ul și anon key-ul tău Supabase și face request-uri direct cu propriul JWT, poate vedea/modifica NUMAI rândurile unde auth.uid() = user_id. Este linia de apărare finală și esențială.
