Integrare Eloquent cu pdphilip/opensearch

Modele Eloquent pe OpenSearch — instalare, CRUD, căutare, sincronizare cu MongoDB

Dacă în Lecțiile 3–6 ai lucrat cu clientul low-level pentru a înțelege Query DSL, acum trecem la nivelul de productivitate: modele Eloquent care trăiesc direct în OpenSearch. Sintaxa va fi familiară — este Eloquent pe care îl știi deja.

1. Instalare

composer require pdphilip/opensearch

Înregistrează Service Provider-ul în bootstrap/providers.php:

// bootstrap/providers.php
return [
    App\Providers\AppServiceProvider::class,
    PDPhilip\OpenSearch\OpenSearchServiceProvider::class,   // adaugă aceasta
];

Configurare conexiune în database.php

// config/database.php — adaugă în array-ul 'connections'
'opensearch' => [
    'driver'       => 'opensearch',
    'hosts'        => explode(',', env('OS_HOSTS', 'http://localhost:9200')),
    'basic_auth'   => [
        'username' => env('OS_USERNAME', ''),
        'password' => env('OS_PASSWORD', ''),
    ],
    'index_prefix' => env('OS_INDEX_PREFIX', false),
    'options'      => [
        'ssl_verification' => env('OS_OPT_VERIFY_SSL', false),
        'default_limit'    => env('OS_OPT_DEFAULT_LIMIT', 1000),
    ],
],

Adaugă în .env:

OS_HOSTS=http://localhost:9200
OS_USERNAME=
OS_PASSWORD=
OS_INDEX_PREFIX=myapp_
OS_OPT_VERIFY_SSL=false
OS_OPT_DEFAULT_LIMIT=1000

2. Crearea unui model OpenSearch

În loc să extindă Illuminate\Database\Eloquent\Model, modelul tău extinde PDPhilip\OpenSearch\Eloquent\Model. Restul funcționează la fel. 

// app/Models/Search/ArticleIndex.php
namespace App\Models\Search;

use PDPhilip\OpenSearch\Eloquent\Model;

class ArticleIndex extends Model
{
    // Conexiunea OpenSearch din database.php
    protected $connection = 'opensearch';

    // Numele indexului (implicit: numele clasei pluralizat snake_case)
    // 'article_indices' → vrei să fie 'articles', deci specifici manual
    protected $index = 'articles';

    // Câmpurile care pot fi asignate în masă
    protected $fillable = [
        'mongo_id',
        'titlu',
        'continut',
        'autor',
        'slug',
        'status',
        'taguri',
        'publicat_la',
        'vizualizari',
        'este_publicat',
    ];

    // Câmpuri care nu apar în JSON (dacă ai date sensibile)
    protected $hidden = [];
}
Convenție recomandată: namespace separat Pune modelele OpenSearch în App\Models\Search\ — le separi clar de modelele MongoDB din App\Models\. Evită confuzia între „modelul operațional" și „modelul de search".

3. CRUD cu Eloquent

Create

// Creare directă
$doc = ArticleIndex::create([
    'mongo_id'    => (string) $article->_id,
    'titlu'       => $article->titlu,
    'continut'    => $article->continut,
    'autor'       => $article->autor,
    'status'      => $article->status,
    'taguri'      => $article->taguri,
    'publicat_la' => $article->created_at,
    'vizualizari' => 0,
]);

Read

// Caută după ID OpenSearch
$doc = ArticleIndex::find($openSearchId);

// Caută după mongo_id (câmpul tău custom)
$doc = ArticleIndex::where('mongo_id', $mongoId)->first();

// Toate documentele
$docs = ArticleIndex::all();

// Cu condiție
$docs = ArticleIndex::where('status', 'published')->get();

Update

// Update simplu
$doc = ArticleIndex::where('mongo_id', $mongoId)->first();
$doc->titlu = $nouTitlu;
$doc->save();

// Update în masă
ArticleIndex::where('status', 'draft')->update(['status' => 'published']);

// Update sau creare (upsert)
ArticleIndex::updateOrCreate(
    ['mongo_id' => (string) $article->_id],   // condiție de match
    [                                           // date de salvat
        'titlu'    => $article->titlu,
        'continut' => $article->continut,
        'status'   => $article->status,
    ]
);

Delete

// Ștergere după instanță
$doc = ArticleIndex::where('mongo_id', $mongoId)->first();
$doc->delete();

// Ștergere directă
ArticleIndex::where('mongo_id', $mongoId)->delete();

4. Căutare cu Eloquent

// whereMatch — căutare full-text în câmp specific
$results = ArticleIndex::whereMatch('titlu', 'laravel')->get();

// Combinat cu filtre Eloquent standard
$results = ArticleIndex
    ::whereMatch('titlu', 'laravel')
    ->where('status', 'published')          // filtru exact
    ->orderBy('publicat_la', 'desc')        // sortare
    ->limit(10)
    ->get();

// whereFuzzy — tolerant la typo-uri
$results = ArticleIndex::whereFuzzy('titlu', 'laravell')->get();

// Paginație nativă Laravel
$results = ArticleIndex
    ::whereMatch('titlu', request('q'))
    ->where('status', 'published')
    ->paginate(15);   // returnează LengthAwarePaginator

5. Sincronizare MongoDB → OpenSearch

Cel mai curat pattern: folosești Eloquent Observers pe modelul MongoDB pentru a sincroniza automat cu OpenSearch la orice modificare. 

// app/Observers/ArticleObserver.php
namespace App\Observers;

use App\Models\Article;           // modelul MongoDB
use App\Models\Search\ArticleIndex; // modelul OpenSearch

class ArticleObserver
{
    // Apelat după Article::create() sau ->save() pe un model nou
    public function created(Article $article): void
    {
        ArticleIndex::create($this->toIndexData($article));
    }

    // Apelat după ->save() pe un model existent
    public function updated(Article $article): void
    {
        ArticleIndex::updateOrCreate(
            ['mongo_id' => (string) $article->_id],
            $this->toIndexData($article)
        );
    }

    // Apelat după ->delete()
    public function deleted(Article $article): void
    {
        ArticleIndex::where('mongo_id', (string) $article->_id)->delete();
    }

    // Extrage doar câmpurile relevante pentru search
    private function toIndexData(Article $article): array
    {
        return [
            'mongo_id'      => (string) $article->_id,
            'titlu'         => $article->titlu,
            'continut'      => $article->continut,
            'autor'         => $article->autor,
            'slug'          => $article->slug,
            'status'        => $article->status,
            'taguri'        => $article->taguri ?? [],
            'publicat_la'   => $article->created_at?->toIso8601String(),
            'este_publicat' => $article->status === 'published',
            'vizualizari'   => $article->vizualizari ?? 0,
        ];
    }
}

Înregistrează observer-ul în AppServiceProvider:

// app/Providers/AppServiceProvider.php
use App\Models\Article;
use App\Observers\ArticleObserver;

public function boot(): void
{
    Article::observe(ArticleObserver::class);
}
De acum înainte Orice Article::create(), $article->save() sau $article->delete() va sincroniza automat OpenSearch. Nu mai trebuie să te gândești la asta în controllere.

6. Re-sincronizare completă (migrare inițială)

// app/Console/Commands/OpenSearchResync.php
class OpenSearchResync extends Command
{
    protected $signature   = 'opensearch:resync';
    protected $description = 'Re-indexează toate articolele din MongoDB în OpenSearch';

    public function handle(): int
    {
        $total = Article::count();
        $bar   = $this->output->createProgressBar($total);

        Article::chunk(500, function ($articles) use ($bar) {
            foreach ($articles as $article) {
                ArticleIndex::updateOrCreate(
                    ['mongo_id' => (string) $article->_id],
                    $this->toIndexData($article)
                );
                $bar->advance();
            }
        });

        $bar->finish();
        $this->newLine();
        $this->info("✓ {$total} articole re-indexate.");
        return 0;
    }

    private function toIndexData(Article $article): array
    {
        return [
            'mongo_id'      => (string) $article->_id,
            'titlu'         => $article->titlu,
            'continut'      => $article->continut,
            'autor'         => $article->autor,
            'slug'          => $article->slug,
            'status'        => $article->status,
            'taguri'        => $article->taguri ?? [],
            'publicat_la'   => $article->created_at?->toIso8601String(),
            'este_publicat' => $article->status === 'published',
            'vizualizari'   => $article->vizualizari ?? 0,
        ];
    }
}

7. Controller complet — de la cerere la răspuns

// app/Http/Controllers/SearchController.php
class SearchController extends Controller
{
    public function index(Request $request)
    {
        $query = $request->get('q', '');
        $tag   = $request->get('tag');

        $builder = ArticleIndex::where('status', 'published');

        if ($query) {
            $builder->whereMatch('titlu', $query)
                     ->orWhereMatch('continut', $query);
        }

        if ($tag) {
            $builder->where('taguri', $tag);
        }

        $results = $builder
            ->orderBy($query ? '_score' : 'publicat_la', 'desc')
            ->paginate(15);

        return response()->json($results);
    }
}
